let map, startMarker, destMarker, routeLine; // 지도 초기화 window.onload = () => { map = L.map('map').setView([37.498095, 127.027610], 14); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { maxZoom: 19, attribution: '© OpenStreetMap' }).addTo(map); }; // 마커 및 경로 그리기 함수 function drawMapRoute(startCoords, destCoords) { if (startMarker) map.removeLayer(startMarker); if (destMarker) map.removeLayer(destMarker); if (routeLine) map.removeLayer(routeLine); const startPoint = [startCoords.lat, startCoords.lng]; const destPoint = [destCoords.lat, destCoords.lng]; startMarker = L.marker(startPoint).addTo(map).bindPopup("출발지"); destMarker = L.marker(destPoint).addTo(map).bindPopup("목적지"); // 초록색 경로선 routeLine = L.polyline([startPoint, destPoint], { color: '#22c55e', weight: 5 }).addTo(map); // 두 지점이 모두 보이도록 지도 영역 맞춤 map.fitBounds(L.latLngBounds([startPoint, destPoint]), { padding: [50, 50] }); }
